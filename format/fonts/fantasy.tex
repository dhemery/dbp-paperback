\def\textsize{11pt}
\def\fineprintsize{10pt}
\font\rm=garamond-11-roman at \textsize
\font\it=garamond-11-italic at \textsize
\font\sc=garamond-11-roman-smallcap at \textsize
\font\fineprint=garamond-11-roman at \fineprintsize

\def\headlinefontfamily{Garamond Premier Pro Caption}
\def\displayfontfamily{Garamond Premier Pro Display}

\newdimen\headlinefontsize \headlinefontsize=8pt
\newdimen\booktitlefontsize \booktitlefontsize=30pt
\newdimen\authornamefontsize \authornamefontsize=24pt
\newdimen\publishernamefontsize \publishernamefontsize=24pt

\newdimen\aheadsize \aheadsize=24pt
\newdimen\bheadsize \bheadsize=14pt

\newdimen\leading \leading=15pt

% Headline and Footline Fonts
\makefont{\headlinefont}{\headlinefontfamily}{\headlinefontsize}{:+cpsp}
\def\footlinefont{\headlinefont}

% Heading and Title Fonts
\makefont{\booktitlefont}{\displayfontfamily}{\booktitlefontsize}{:+smcp+cpsp}
\makefont{\authornamefont}{\displayfontfamily}{\authornamefontsize}{:+smcp+cpsp}
\makefont{\chaptertitlefont}{\displayfontfamily}{\aheadsize}{:+smcp}
\makefont{\scenetitlefont}{\displayfontfamily}{\bheadsize}{:+smcp}
\def\bookinfotitlefont{\scenetitlefont}
\makefont{\authorinfotitlefont}{\displayfontfamily}{\fineprintsize}{:+cpsp}

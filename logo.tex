\def\linelogo#1{
  \begingroup
  \dimen0=#1\divide\dimen0 by 16
  \font\logofont="Proxima Nova:+cpsp" at #1\relax
  \logofont\baselineskip=0pt
  \vbox{
    \hrule height \dimen0
    \vskip2\dimen0
    \hbox{DRISCOLL BROOK PRESS}
    \vskip2\dimen0
    \hrule height \dimen0
  }
  \endgroup
}

\def\stackedlogo#1{
  \begingroup
  \dimen0=#1\divide\dimen0 by 16
  \font\logofont="Proxima Nova:+cpsp" at #1\relax
  \logofont\baselineskip=0pt
  \setbox0=\hbox{DRISCOLL}
  \vbox{
    \hrule height \dimen0
    \vskip2\dimen0
    \copy0
    \vskip2\dimen0
    \hbox to \wd0{\hss BROOK}
    \vskip2\dimen0
    \hbox to \wd0{\hss PRESS}
    \vskip2\dimen0
    \hrule height \dimen0
  }
  \endgroup
}

\rightline{\linelogo{16pt}}\par

\rightline{\stackedlogo{30pt}}\par

\bye

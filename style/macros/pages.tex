\def\clearpage{\eject}

% TODO Make this work
\def\cleartorecto{\clearpage\clearpage}

% \titlepage{title}{author}{publisher}
\def\titlepage{\maketitlepage{\title}{\author}{\publisher}}

% \halftitlepage{title}
\def\halftitlepage{\maketitlepage{\title}{}{}}

% An info page is a frontmatter or backmatter page
% other than a title or halftitle page.
\def\infopage#1{
  \cleartorecto
  \infopageany{#1}
}

\def\infopageany#1{
  \clearpage
  \infopagetitle{#1}
  \bigskip
}

\def\maketitlepage#1#2#3{% title, author, publisher
  \cleartorecto
  {
    \parindent=0pt
    \bigskip
    \titlepagetitle{#1}
    \bigskip
    \titlepageauthor{#2}
    \vfill
    \titlepagepublisher{#3}
  }
  \clearpage
}

\def\titlepagetitle#1{{\titlepagetitlestyle\rightline{#1}}}
\def\titlepagetitlestyle{\titlepagetitlefont}
\def\titlepageauthor#1{{\titlepageauthorstyle\rightline{#1}}}
\def\titlepageauthorstyle{\titlepageauthorfont}
\def\titlepagepublisher#1{{\titlepagepublisherstyle\rightline{#1}}}
\def\titlepagepublisherstyle{\titlepagepublisherfont}

\def\infopagetitle#1{{\noindent\infopagetitlestyle\leftline{#1}}}
\def\infopagetitlestyle{\infopagetitlefont}

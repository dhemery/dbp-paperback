\def\clearpage{\eject}
\def\cleartorecto{\clearpage\ifodd\pageno \else\thispagestyle{cleared}\null\clearpage\fi}

% \titlepage{title}{author}{publisher}
\def\titlepage{\maketitlepage{\title}{\author}{\publisher}}

% \halftitlepage{title}
\def\halftitlepage{\maketitlepage{\title}{}{}}

% An info page is a frontmatter or backmatter page
% other than a title or halftitle page.
\def\infopage#1{\cleartorecto\infopageany{#1}}
\def\infopageany#1{\clearpage\infopagetitle{#1}}

\def\maketitlepage#1#2#3{% title, author, publisher
  \cleartorecto
  \titlepagetitle{#1}\titlepageauthor{#2}\titlepagepublisher{#3}
  \clearpage
}

\def\titlepagetitle#1{{\titlepagetitlestyle #1}\par}
\def\titlepagetitlestyle{\noindent\titlepagetitlefont}
\def\titlepageauthor#1{{\titlepageauthorstyle #1}\par}
\def\titlepageauthorstyle{\noindent\titlepageauthorfont}
\def\titlepagepublisher#1{{\titlepagepublisherstyle #1}\par}
\def\titlepagepublisherstyle{\noindent\titlepagepublisherfont}

\def\infopagetitle#1{{\infopagetitlestyle #1}\par}
\def\infopagetitlestyle{\noindent\infopagetitlefont}

\def\frontmatter{
  \cleartorecto
  \pagestyle{empty}
}
\def\mainmatter{
  \cleartorecto
  \pagestyle{normal}
}
\def\backmatter{
  \cleartorecto
  \pagestyle{empty}
}

\gdef\psthis{}
\gdef\psnext{}

\def\thispagestyle#1{
  \gdef\psthis{#1}
  \headline{headline:\hfil\psthis}
  \footline{
    footline:\hfil\psthis
    \thispagestyle{\psnext}
  }
}
\def\pagestyle#1{
  \gdef\psnext{#1}
  \thispagestyle{#1}
}

\def\clearpage{\eject}
\def\cleartorecto{\clearpage\clearpage}

\def\halftitlepage{\maketitlepage{\title}{}{}}
\def\titlepage{\maketitlepage{\title}{\author}{\publisher}}

\def\maketitlepage#1#2#3{% title, author, publisher
  \cleartorecto
  {
    \parindent=0pt
    \titlepagetitle{\bigskip#1}
    \titlepageauthor{\bigskip#2}
    \titlepagepublisher{\vfill#3}
  }
  \clearpage
}
\def\titlepagetitle#1{{\titlepagetitlefont#1}}
\def\titlepageauthor#1{{\titlepageauthorfont#1}}
\def\titlepagepublisher#1{{\titlepagepublisherfont#1}}
\def\titlepagetitlefont{\bf}
\def\titlepageauthorfont{\it}
\def\titlepagepublisherfont{\it\bf}


% An info page is a frontmatter or backmatter page
% other than a title or halftitle page.
\def\infopage#1{
  \cleartorecto
  \infopageany{#1}
}
\def\infopageany#1{
  \clearpage
  \infopagetitle{#1}
  \bigskip
}
\def\infopagetitle#1{{\noindent\infopagetitlefont#1\par}}
\def\infopagetitlefont{\bf}

% An about page is an info page that appears in the toc.
% TODO Add page to toc.
\def\aboutpage#1{\infopage{#1}}

% Scene opener page
\def\scene#1{
  \clearforscene
  \thispagestyle{opener}
  \scenetitle{#1}
  \noindent
}
\def\scenetitle#1{{\noindent\scenetitlefont#1\par}}
\def\clearforscene{\cleartorecto}
\def\scenetitlefont{\bf}

% Items on info pages
\def\infoitem#1{{\infoitemfont#1\par}}
\def\infoitemtitle#1{{\infoitemtitlefont#1\smallskip}}
\def\infoitemtitlefont{\bf}
\def\infoitemfont{\it}
